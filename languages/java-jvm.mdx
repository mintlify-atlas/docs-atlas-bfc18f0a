<Columns cols={2}>
  <Card title="MLlib (Apache Spark)" icon="fire" href="https://spark.apache.org/docs/latest/mllib-guide.html">

  </Card>
  <Card title="Hydrosphere Mist" icon="cloud" href="https://github.com/Hydrospheredata/mist">

  </Card>
  <Card title="ORYX" icon="arrows-spin" href="https://github.com/oryxproject/oryx">

  </Card>
  <Card title="Apache Mahout" icon="elephant" href="https://github.com/apache/mahout">
